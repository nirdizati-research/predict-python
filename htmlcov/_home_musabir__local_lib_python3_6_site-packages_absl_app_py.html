181def _register_and_parse_flags_with_usage( 

182 argv=None, 

183 flags_parser=parse_flags_with_usage, 

184): 

185 """Registers help flags, parses arguments and shows usage if appropriate. 

186 

187 This also calls sys.exit(0) if flag --only_check_args is True. 

188 

189 Args: 

190 argv: [str], a non-empty list of the command line arguments including 

191 program name, sys.argv is used if None. 

192 flags_parser: Callable[[List[Text]], Any], the function used to parse flags. 

193 The return value of this function is passed to `main` untouched. 

194 It must guarantee FLAGS is parsed after this function is called. 

195 

196 Returns: 

197 The return value of `flags_parser`. When using the default `flags_parser`, 

198 it returns the following: 

199 [str], a non-empty list of remaining command line arguments after parsing 

200 flags, including program name. 

201 

202 Raises: 

203 Error: Raised when flags_parser is called, but FLAGS is not parsed. 

204 SystemError: Raised when it's called more than once. 

205 """ 

206 if _register_and_parse_flags_with_usage.done: 

207 raise SystemError('Flag registration can be done only once.') 

208 

209 define_help_flags() 

210 

211 original_argv = sys.argv if argv is None else argv 

212 args_to_main = flags_parser(original_argv) 

213 if not FLAGS.is_parsed(): 

214 raise Error('FLAGS must be parsed after flags_parser is called.') 

215 

216 # Exit when told so. 

217 if FLAGS.only_check_args: 

218 sys.exit(0) 

219 # Immediately after flags are parsed, bump verbosity to INFO if the flag has 

220 # not been set. 

221 if FLAGS['verbosity'].using_default_value: 

222 FLAGS.verbosity = 0 

223 _register_and_parse_flags_with_usage.done = True 

224

230def _run_main(main, argv): 

231 """Calls main, optionally with pdb or profiler.""" 

232 if FLAGS.run_with_pdb: 

233 sys.exit(pdb.runcall(main, argv)) 

234 elif FLAGS.run_with_profiling or FLAGS.profile_file: 

235 # Avoid import overhead since most apps (including performance-sensitive 

236 # ones) won't be run with profiling. 

237 import atexit 

238 if FLAGS.use_cprofile_for_profiling: 

239 import cProfile as profile 

240 else: 

241 import profile 

242 profiler = profile.Profile() 

243 if FLAGS.profile_file: 

244 atexit.register(profiler.dump_stats, FLAGS.profile_file) 

245 else: 

246 atexit.register(profiler.print_stats) 

247 retval = profiler.runcall(main, argv) 

248 sys.exit(retval) 

249 else: 

250 sys.exit(main(argv))

253def _call_exception_handlers(exception): 

254 """Calls any installed exception handlers.""" 

255 for handler in EXCEPTION_HANDLERS: 

256 try: 

257 if handler.wants(exception): 

258 handler.handle(exception) 

259 except: # pylint: disable=bare-except 

260 try: 

261 # We don't want to stop for exceptions in the exception handlers but 

262 # we shouldn't hide them either. 

263 logging.error(traceback.format_exc()) 

264 except: # pylint: disable=bare-except 

265 # In case even the logging statement fails, ignore. 

266 pass

269def run( 

270 main, 

271 argv=None, 

272 flags_parser=parse_flags_with_usage, 

273): 

274 """Begins executing the program. 

275 

276 Args: 

277 main: The main function to execute. It takes an single argument "argv", 

278 which is a list of command line arguments with parsed flags removed. 

279 If it returns an integer, it is used as the process's exit code. 

280 argv: A non-empty list of the command line arguments including program name, 

281 sys.argv is used if None. 

282 flags_parser: Callable[[List[Text]], Any], the function used to parse flags. 

283 The return value of this function is passed to `main` untouched. 

284 It must guarantee FLAGS is parsed after this function is called. 

285 - Parses command line flags with the flag module. 

286 - If there are any errors, prints usage(). 

287 - Calls main() with the remaining arguments. 

288 - If main() raises a UsageError, prints usage and the error message. 

289 """ 

290 try: 

291 args = _run_init( 

292 sys.argv if argv is None else argv, 

293 flags_parser, 

294 ) 

295 while _init_callbacks: 

296 callback = _init_callbacks.popleft() 

297 callback() 

298 try: 

299 _run_main(main, args) 

300 except UsageError as error: 

301 usage(shorthelp=True, detailed_error=error, exitcode=error.exitcode) 

302 except: 

303 exc = sys.exc_info()[1] 

304 # Don't try to post-mortem debug successful SystemExits, since those 

305 # mean there wasn't actually an error. In particular, the test framework 

306 # raises SystemExit(False) even if all tests passed. 

307 if isinstance(exc, SystemExit) and not exc.code: 

308 raise 

309 

310 # Check the tty so that we don't hang waiting for input in an 

311 # non-interactive scenario. 

312 if FLAGS.pdb_post_mortem and sys.stdout.isatty(): 

313 traceback.print_exc() 

314 print() 

315 print(' *** Entering post-mortem debugging ***') 

316 print() 

317 pdb.post_mortem() 

318 raise 

319 except Exception as e: 

320 _call_exception_handlers(e) 

321 raise

327def call_after_init(callback): 

328 """Calls the given callback only once ABSL has finished initialization. 

329 

330 If ABSL has already finished initialization when `call_after_init` is 

331 called then the callback is executed immediately, otherwise `callback` is 

332 stored to be executed after `app.run` has finished initializing (aka. just 

333 before the main function is called). 

334 

335 If called after `app.run`, this is equivalent to calling `callback()` in the 

336 caller thread. If called before `app.run`, callbacks are run sequentially (in 

337 an undefined order) in the same thread as `app.run`. 

338 

339 Args: 

340 callback: a callable to be called once ABSL has finished initialization. 

341 This may be immediate if initialization has already finished. It 

342 takes no arguments and returns nothing. 

343 """ 

344 if _run_init.done: 

345 callback() 

346 else: 

347 _init_callbacks.append(callback)

350def _run_init( 

351 argv, 

352 flags_parser, 

353): 

354 """Does one-time initialization and re-parses flags on rerun.""" 

355 if _run_init.done: 

356 return flags_parser(argv) 

357 command_name.make_process_name_useful() 

358 # Set up absl logging handler. 

359 logging.use_absl_handler() 

360 args = _register_and_parse_flags_with_usage( 

361 argv=argv, 

362 flags_parser=flags_parser, 

363 ) 

364 if faulthandler: 

365 try: 

366 faulthandler.enable() 

367 except Exception: # pylint: disable=broad-except 

368 # Some tests verify stderr output very closely, so don't print anything.

370 pass 

371 _run_init.done = True

378def usage(shorthelp=False, writeto_stdout=False, detailed_error=None, 

379 exitcode=None): 

380 """Writes __main__'s docstring to stderr with some help text. 

381 

382 Args: 

383 shorthelp: bool, if True, prints only flags from the main module, 

384 rather than all flags. 

385 writeto_stdout: bool, if True, writes help message to stdout, 

386 rather than to stderr. 

387 detailed_error: str, additional detail about why usage info was presented. 

388 exitcode: optional integer, if set, exits with this status code after 

389 writing help. 

390 """ 

391 if writeto_stdout: 

392 stdfile = sys.stdout 

393 else: 

394 stdfile = sys.stderr 

395 

396 doc = sys.modules['__main__'].__doc__ 

397 if not doc: 

398 doc = '\nUSAGE: %s [flags]\n' % sys.argv[0] 

399 doc = flags.text_wrap(doc, indent=' ', firstline_indent='') 

400 else: 

401 # Replace all '%s' with sys.argv[0], and all '%%' with '%'. 

402 num_specifiers = doc.count('%') - 2 * doc.count('%%') 

403 try: 

404 doc %= (sys.argv[0],) * num_specifiers 

405 except (OverflowError, TypeError, ValueError): 

406 # Just display the docstring as-is. 

407 pass 

408 if shorthelp: 

409 flag_str = FLAGS.main_module_help() 

410 else: 

411 flag_str = FLAGS.get_help() 

412 try: 

413 stdfile.write(doc) 

414 if flag_str: 

415 stdfile.write('\nflags:\n') 

416 stdfile.write(flag_str) 

417 stdfile.write('\n') 

418 if detailed_error is not None: 

419 stdfile.write('\n%s\n' % detailed_error) 

420 except IOError as e: 

421 # We avoid printing a huge backtrace if we get EPIPE, because 

422 # "foo.par --help | less" is a frequent use case. 

423 if e.errno != errno.EPIPE: 

424 raise 

425 if exitcode is not None: 

426 sys.exit(exitcode)
