42 def get_table_list(self, cursor): 

43 """Return a list of table and view names in the current database.""" 

44 cursor.execute(""" 

45 SELECT c.relname, 

46 CASE WHEN {} THEN 'p' WHEN c.relkind IN ('m', 'v') THEN 'v' ELSE 't' END 

47 FROM pg_catalog.pg_class c 

48 LEFT JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace 

49 WHERE c.relkind IN ('f', 'm', 'p', 'r', 'v') 

50 AND n.nspname NOT IN ('pg_catalog', 'pg_toast') 

51 AND pg_catalog.pg_table_is_visible(c.oid) 

52 """.format('c.relispartition' if self.connection.features.supports_table_partitions else 'FALSE')) 

53 return [TableInfo(*row) for row in cursor.fetchall() if row[0] not in self.ignored_tables]

55 def get_table_description(self, cursor, table_name): 

56 """ 

57 Return a description of the table with the DB-API cursor.description 

58 interface. 

59 """ 

60 # Query the pg_catalog tables as cursor.description does not reliably 

61 # return the nullable property and information_schema.columns does not 

62 # contain details of materialized views. 

63 cursor.execute(""" 

64 SELECT 

65 a.attname AS column_name, 

66 NOT (a.attnotnull OR (t.typtype = 'd' AND t.typnotnull)) AS is_nullable, 

67 pg_get_expr(ad.adbin, ad.adrelid) AS column_default 

68 FROM pg_attribute a 

69 LEFT JOIN pg_attrdef ad ON a.attrelid = ad.adrelid AND a.attnum = ad.adnum 

70 JOIN pg_type t ON a.atttypid = t.oid 

71 JOIN pg_class c ON a.attrelid = c.oid 

72 JOIN pg_namespace n ON c.relnamespace = n.oid 

73 WHERE c.relkind IN ('f', 'm', 'p', 'r', 'v') 

74 AND c.relname = %s 

75 AND n.nspname NOT IN ('pg_catalog', 'pg_toast') 

76 AND pg_catalog.pg_table_is_visible(c.oid) 

77 """, [table_name]) 

78 field_map = {line[0]: line[1:] for line in cursor.fetchall()} 

79 cursor.execute("SELECT * FROM %s LIMIT 1" % self.connection.ops.quote_name(table_name)) 

80 return [ 

81 FieldInfo( 

82 line.name, 

83 line.type_code, 

84 line.display_size, 

85 line.internal_size, 

86 line.precision, 

87 line.scale, 

88 *field_map[line.name], 

89 ) 

90 for line in cursor.description 

91 ]

93 def get_sequences(self, cursor, table_name, table_fields=()): 

94 cursor.execute(""" 

95 SELECT s.relname as sequence_name, col.attname 

96 FROM pg_class s 

97 JOIN pg_namespace sn ON sn.oid = s.relnamespace 

98 JOIN pg_depend d ON d.refobjid = s.oid AND d.refclassid = 'pg_class'::regclass 

99 JOIN pg_attrdef ad ON ad.oid = d.objid AND d.classid = 'pg_attrdef'::regclass 

100 JOIN pg_attribute col ON col.attrelid = ad.adrelid AND col.attnum = ad.adnum 

101 JOIN pg_class tbl ON tbl.oid = ad.adrelid 

102 JOIN pg_namespace n ON n.oid = tbl.relnamespace 

103 WHERE s.relkind = 'S' 

104 AND d.deptype in ('a', 'n') 

105 AND n.nspname = 'public' 

106 AND tbl.relname = %s 

107 """, [table_name]) 

108 return [ 

109 {'name': row[0], 'table': table_name, 'column': row[1]} 

110 for row in cursor.fetchall()

113 def get_relations(self, cursor, table_name): 

114 """ 

115 Return a dictionary of {field_name: (field_name_other_table, other_table)} 

116 representing all relationships to the given table. 

117 """ 

118 cursor.execute(""" 

119 SELECT c2.relname, a1.attname, a2.attname 

120 FROM pg_constraint con 

121 LEFT JOIN pg_class c1 ON con.conrelid = c1.oid 

122 LEFT JOIN pg_class c2 ON con.confrelid = c2.oid 

123 LEFT JOIN pg_attribute a1 ON c1.oid = a1.attrelid AND a1.attnum = con.conkey[1] 

124 LEFT JOIN pg_attribute a2 ON c2.oid = a2.attrelid AND a2.attnum = con.confkey[1] 

125 WHERE c1.relname = %s AND con.contype = 'f' 

126 """, [table_name]) 

127 return {row[1]: (row[2], row[0]) for row in cursor.fetchall()}

129 def get_key_columns(self, cursor, table_name): 

130 cursor.execute(""" 

131 SELECT kcu.column_name, ccu.table_name AS referenced_table, ccu.column_name AS referenced_column 

132 FROM information_schema.constraint_column_usage ccu 

133 LEFT JOIN information_schema.key_column_usage kcu 

134 ON ccu.constraint_catalog = kcu.constraint_catalog 

135 AND ccu.constraint_schema = kcu.constraint_schema 

136 AND ccu.constraint_name = kcu.constraint_name 

137 LEFT JOIN information_schema.table_constraints tc 

138 ON ccu.constraint_catalog = tc.constraint_catalog 

139 AND ccu.constraint_schema = tc.constraint_schema 

140 AND ccu.constraint_name = tc.constraint_name 

141 WHERE kcu.table_name = %s AND tc.constraint_type = 'FOREIGN KEY' 

142 """, [table_name]) 

143 return cursor.fetchall()

145 def get_constraints(self, cursor, table_name): 

146 """ 

147 Retrieve any constraints or keys (unique, pk, fk, check, index) across 

148 one or more columns. Also retrieve the definition of expression-based 

149 indexes. 

150 """ 

151 constraints = {} 

152 # Loop over the key table, collecting things as constraints. The column 

153 # array must return column names in the same order in which they were 

154 # created. 

155 cursor.execute(""" 

156 SELECT 

157 c.conname, 

158 array( 

159 SELECT attname 

160 FROM unnest(c.conkey) WITH ORDINALITY cols(colid, arridx) 

161 JOIN pg_attribute AS ca ON cols.colid = ca.attnum 

162 WHERE ca.attrelid = c.conrelid 

163 ORDER BY cols.arridx 

164 ), 

165 c.contype, 

166 (SELECT fkc.relname || '.' || fka.attname 

167 FROM pg_attribute AS fka 

168 JOIN pg_class AS fkc ON fka.attrelid = fkc.oid 

169 WHERE fka.attrelid = c.confrelid AND fka.attnum = c.confkey[1]), 

170 cl.reloptions 

171 FROM pg_constraint AS c 

172 JOIN pg_class AS cl ON c.conrelid = cl.oid 

173 JOIN pg_namespace AS ns ON cl.relnamespace = ns.oid 

174 WHERE ns.nspname = %s AND cl.relname = %s 

175 """, ["public", table_name]) 

176 for constraint, columns, kind, used_cols, options in cursor.fetchall(): 

177 constraints[constraint] = { 

178 "columns": columns, 

179 "primary_key": kind == "p", 

180 "unique": kind in ["p", "u"], 

181 "foreign_key": tuple(used_cols.split(".", 1)) if kind == "f" else None, 

182 "check": kind == "c", 

183 "index": False, 

184 "definition": None, 

185 "options": options, 

186 } 

187 # Now get indexes 

188 cursor.execute(""" 

189 SELECT 

190 indexname, array_agg(attname ORDER BY arridx), indisunique, indisprimary, 

191 array_agg(ordering ORDER BY arridx), amname, exprdef, s2.attoptions 

192 FROM ( 

193 SELECT 

194 c2.relname as indexname, idx.*, attr.attname, am.amname, 

195 CASE 

196 WHEN idx.indexprs IS NOT NULL THEN 

197 pg_get_indexdef(idx.indexrelid) 

198 END AS exprdef, 

199 CASE am.amname 

200 WHEN 'btree' THEN 

201 CASE (option & 1) 

202 WHEN 1 THEN 'DESC' ELSE 'ASC' 

203 END 

204 END as ordering, 

205 c2.reloptions as attoptions 

206 FROM ( 

207 SELECT * 

208 FROM pg_index i, unnest(i.indkey, i.indoption) WITH ORDINALITY koi(key, option, arridx) 

209 ) idx 

210 LEFT JOIN pg_class c ON idx.indrelid = c.oid 

211 LEFT JOIN pg_class c2 ON idx.indexrelid = c2.oid 

212 LEFT JOIN pg_am am ON c2.relam = am.oid 

213 LEFT JOIN pg_attribute attr ON attr.attrelid = c.oid AND attr.attnum = idx.key 

214 WHERE c.relname = %s 

215 ) s2 

216 GROUP BY indexname, indisunique, indisprimary, amname, exprdef, attoptions; 

217 """, [table_name]) 

218 for index, columns, unique, primary, orders, type_, definition, options in cursor.fetchall(): 

219 if index not in constraints: 

220 basic_index = type_ == 'btree' and not index.endswith('_btree') and options is None 

221 constraints[index] = { 

222 "columns": columns if columns != [None] else [], 

223 "orders": orders if orders != [None] else [], 

224 "primary_key": primary, 

225 "unique": unique, 

226 "foreign_key": None, 

227 "check": False, 

228 "index": True, 

229 "type": Index.suffix if basic_index else type_, 

230 "definition": definition, 

231 "options": options, 

232 } 

233 return constraints
