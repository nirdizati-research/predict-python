17def get_commit_mode(): 

18 """ 

19 Disabling AUTOCOMMIT causes enqueued jobs to be stored in a temporary queue. 

20 Jobs in this queue are only enqueued after the request is completed and are 

21 discarded if the request causes an exception (similar to db transactions). 

22 

23 To disable autocommit, put this in settings.py: 

24 RQ = { 

25 'AUTOCOMMIT': False, 

26 } 

27 """ 

28 RQ = getattr(settings, 'RQ', {}) 

29 return RQ.get('AUTOCOMMIT', True)

53class DjangoRQ(Queue): 

54 """ 

55 A subclass of RQ's QUEUE that allows jobs to be stored temporarily to be 

56 enqueued later at the end of Django's request/response cycle. 

57 """ 

58

60 autocommit = kwargs.pop('autocommit', None) 

61 self._autocommit = get_commit_mode() if autocommit is None else autocommit 

62 

63 super(DjangoRQ, self).__init__(*args, **kwargs)

65 def original_enqueue_call(self, *args, **kwargs): 

66 return super(DjangoRQ, self).enqueue_call(*args, **kwargs)

76 """ 

77 Returns a redis connection from a connection config 

78 """

95 # We're using django-redis-cache

107 'db': config.get('DB'),

109 'socket_timeout': config.get('SOCKET_TIMEOUT'),

111 sentinel_kwargs.update(config.get('CONNECTION_KWARGS', {})) 

112 sentinel = Sentinel(config['SENTINELS'], **sentinel_kwargs) 

113 return sentinel.master_for( 

114 service_name=config['MASTER_NAME'], redis_class=redis_cls,

181def get_queues(*queue_names, **kwargs): 

182 """ 

183 Return queue instances from specified queue names. 

184 All instances must use the same Redis connection. 

185 """ 

186 from .settings import QUEUES 

187 

188 if len(queue_names) <= 1: 

189 # Return "default" queue if no queue name is specified 

190 # or one queue with specified name 

191 return [get_queue(*queue_names, **kwargs)] 

192 

193 # will return more than one queue 

194 # import job class only once for all queues 

195 kwargs['job_class'] = get_job_class(kwargs.pop('job_class', None)) 

196 

197 queue_params = QUEUES[queue_names[0]] 

198 connection_params = filter_connection_params(queue_params) 

199 queues = [get_queue(queue_names[0], **kwargs)] 

200 

201 # do consistency checks while building return list 

202 for name in queue_names[1:]: 

203 queue = get_queue(name, **kwargs) 

204 if type(queue) is not type(queues[0]): 

205 raise ValueError( 

206 'Queues must have the same class.' 

207 '"{0}" and "{1}" have ' 

208 'different classes'.format(name, queue_names[0])) 

209 if connection_params != filter_connection_params(QUEUES[name]): 

210 raise ValueError( 

211 'Queues must have the same redis connection.' 

212 '"{0}" and "{1}" have ' 

213 'different connections'.format(name, queue_names[0])) 

214 queues.append(queue) 

215 

216 return queues
