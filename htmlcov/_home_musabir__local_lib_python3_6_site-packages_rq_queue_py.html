423 @classmethod 

424 def lpop(cls, queue_keys, timeout, connection=None): 

425 """Helper method. Intermediate method to abstract away from some 

426 Redis API details, where LPOP accepts only a single key, whereas BLPOP 

427 accepts multiple. So if we want the non-blocking LPOP, we need to 

428 iterate over all queues, do individual LPOPs, and return the result. 

429 

430 Until Redis receives a specific method for this, we'll have to wrap it 

431 this way. 

432 

433 The timeout parameter is interpreted as follows: 

434 None - non-blocking (return immediately) 

435 > 0 - maximum number of seconds to block 

436 """ 

437 connection = resolve_connection(connection) 

438 if timeout is not None: # blocking variant 

439 if timeout == 0: 

440 raise ValueError('RQ does not support indefinite timeouts. Please pick a timeout value > 0') 

441 result = connection.blpop(queue_keys, timeout) 

442 if result is None: 

443 raise DequeueTimeout(timeout, queue_keys) 

444 queue_key, job_id = result 

445 return queue_key, job_id 

446 else: # non-blocking variant 

447 for queue_key in queue_keys: 

448 blob = connection.lpop(queue_key) 

449 if blob is not None: 

450 return queue_key, blob 

451 return None

454 def dequeue_any(cls, queues, timeout, connection=None, job_class=None): 

455 """Class method returning the job_class instance at the front of the given 

456 set of Queues, where the order of the queues is important. 

457

459 either blocks execution of this function for the duration of the

461 None. 

462 

463 See the documentation of cls.lpop for the interpretation of timeout. 

464 """ 

465 job_class = backend_class(cls, 'job_class', override=job_class)
