92 class __dstr(str): 

93 """Compare helper for compatibility with old third-party code. 

94 

95 Old third-party code might still compare the relation with the 

96 previously used relations (<<,<=,==,!=,>=,>>,) instead of the curently 

97 used ones (<,<=,=,!=,>=,>,). This compare helper lets < match to <<, 

98 > match to >> and = match to ==. 

99 """ 

100 

101 def __eq__(self, other): 

102 if str.__eq__(self, other): 

103 return True 

104 elif str.__eq__(self, '<'): 

105 return str.__eq__('<<', other) 

106 elif str.__eq__(self, '>'): 

107 return str.__eq__('>>', other) 

108 elif str.__eq__(self, '='): 

109 return str.__eq__('==', other) 

110 else: 

111 return False 

112 

113 def __ne__(self, other): 

114 return not self.__eq__(other)

441 """Compares against another apt.Version object or a version string.

443 This method behaves like Python 2's cmp builtin and returns an integer

445 self < other, zero if self == other and positive if self > other. 

446 

447 The comparison includes the package name and architecture if other is

592 @property

594 # type: () -> str 

595 """Return the formatted long description. 

596 

597 Return the formatted long description according to the Debian policy 

598 (Chapter 5.6.13). 

599 See http://www.debian.org/doc/debian-policy/ch-controlfields.html 

600 for more information. 

601 """ 

602 desc = '' 

603 records = self._translated_records 

604 dsc = records.long_desc if records is not None else None 

605 

606 if not dsc: 

607 return _("Missing description for '%s'." 

608 "Please report.") % (self.package.name) 

609 

610 try: 

611 if not isinstance(dsc, unicode): 

612 # Only convert where needed (i.e. Python 2.X) 

613 dsc = dsc.decode("utf-8") 

614 except UnicodeDecodeError as err: 

615 return _("Invalid unicode in description for '%s' (%s). " 

616 "Please report.") % (self.package.name, err) 

617 

618 lines = iter(dsc.split("\n")) 

619 # Skip the first line, since its a duplication of the summary 

620 next(lines) 

621 for raw_line in lines: 

622 if raw_line.strip() == ".": 

623 # The line is just line break 

624 if not desc.endswith("\n"): 

625 desc += "\n\n" 

626 continue 

627 if raw_line.startswith(" "): 

628 # The line should be displayed verbatim without word wrapping 

629 if not desc.endswith("\n"): 

630 line = "\n%s\n" % raw_line[2:] 

631 else: 

632 line = "%s\n" % raw_line[2:] 

633 elif raw_line.startswith(" "): 

634 # The line is part of a paragraph. 

635 if desc.endswith("\n") or desc == "": 

636 # Skip the leading white space 

637 line = raw_line[1:] 

638 else: 

639 line = raw_line 

640 else: 

641 line = raw_line 

642 # Add current line to the description 

643 desc += line 

644 return desc

862 def fetch_source(self, destdir="", progress=None, unpack=True): 

863 # type: (str, AcquireProgress, bool) -> str 

864 """Get the source code of a package. 

865 

866 The parameter *destdir* specifies the directory where the source will 

867 be fetched to. 

868 

869 The parameter *progress* may refer to an apt_pkg.AcquireProgress() 

870 object. If not specified or None, apt.progress.text.AcquireProgress() 

871 is used. 

872 

873 The parameter *unpack* describes whether the source should be unpacked 

874 (``True``) or not (``False``). By default, it is unpacked. 

875 

876 If *unpack* is ``True``, the path to the extracted directory is 

877 returned. Otherwise, the path to the .dsc file is returned. 

878 """ 

879 src = apt_pkg.SourceRecords() 

880 acq = apt_pkg.Acquire(progress or apt.progress.text.AcquireProgress()) 

881 

882 dsc = None 

883 record = self._records 

884 source_name = record.source_pkg or self.package.shortname 

885 source_version = record.source_ver or self._cand.ver_str 

886 source_lookup = src.lookup(source_name) 

887 

888 while source_lookup and source_version != src.version: 

889 source_lookup = src.lookup(source_name) 

890 if not source_lookup: 

891 raise ValueError("No source for %r" % self) 

892 files = list() 

893 for md5, size, path, type_ in src.files: 

894 base = os.path.basename(path) 

895 destfile = os.path.join(destdir, base) 

896 if type_ == 'dsc': 

897 dsc = destfile 

898 if _file_is_same(destfile, size, md5): 

899 logging.debug('Ignoring already existing file: %s' % destfile) 

900 continue 

901 files.append(apt_pkg.AcquireFile(acq, src.index.archive_uri(path), 

902 md5, size, base, destfile=destfile)) 

903 acq.run() 

904 

905 if dsc is None: 

906 raise ValueError("No source for %r" % self) 

907 

908 for item in acq.items: 

909 if item.status != item.STAT_DONE: 

910 raise FetchError("The item %r could not be fetched: %s" % 

911 (item.destfile, item.error_text)) 

912 

913 if unpack: 

914 outdir = src.package + '-' + apt_pkg.upstream_version(src.version) 

915 outdir = os.path.join(destdir, outdir) 

916 subprocess.check_call(["dpkg-source", "-x", dsc, outdir]) 

917 return os.path.abspath(outdir) 

918 else: 

919 return os.path.abspath(dsc)

922class VersionList(Sequence): 

923 """Provide a mapping & sequence interface to all versions of a package. 

924 

925 This class can be used like a dictionary, where version strings are the 

926 keys. It can also be used as a sequence, where integers are the keys. 

927 

928 You can also convert this to a dictionary or a list, using the usual way 

929 of dict(version_list) or list(version_list). This is useful if you need 

930 to access the version objects multiple times, because they do not have to 

931 be recreated this way. 

932 

933 Examples ('package.versions' being a version list): 

934 '0.7.92' in package.versions # Check whether 0.7.92 is a valid version. 

935 package.versions[0] # Return first version or raise IndexError 

936 package.versions[0:2] # Return a new VersionList for objects 0-2 

937 package.versions['0.7.92'] # Return version 0.7.92 or raise KeyError 

938 package.versions.keys() # All keys, as strings. 

939 max(package.versions) 

940 """ 

941 

942 def __init__(self, package, slice_=None): 

943 # type: (Package, Any) -> None 

944 self._package = package # apt.package.Package() 

945 self._versions = package._pkg.version_list # [apt_pkg.Version(), ...] 

946 if slice_: 

947 self._versions = self._versions[slice_] 

948 

949 def __getitem__(self, item): 

950 # FIXME: add type hints 

951 if isinstance(item, slice): 

952 return self.__class__(self._package, item) 

953 try: 

954 # Sequence interface, item is an integer 

955 return Version(self._package, self._versions[item]) 

956 except TypeError: 

957 # Dictionary interface item is a string. 

958 for ver in self._versions: 

959 if ver.ver_str == item: 

960 return Version(self._package, ver) 

961 raise KeyError("Version: %r not found." % (item))

974 return (Version(self._package, ver) for ver in self._versions)

977 # type: (object) -> bool 

978 if isinstance(item, Version): # Sequence interface 

979 item = item.version

982 if ver.ver_str == item:

1230 def get_changelog(self, uri=None, cancel_lock=None): 

1231 # type: (str, threading.Event) -> str 

1232 """ 

1233 Download the changelog of the package and return it as unicode 

1234 string. 

1235 

1236 The parameter *uri* refers to the uri of the changelog file. It may 

1237 contain multiple named variables which will be substitued. These 

1238 variables are (src_section, prefix, src_pkg, src_ver). An example is 

1239 the Ubuntu changelog:: 

1240 

1241 "http://changelogs.ubuntu.com/changelogs/pool" \\ 

1242 "/%(src_section)s/%(prefix)s/%(src_pkg)s" \\ 

1243 "/%(src_pkg)s_%(src_ver)s/changelog"

1257 "/%(src_section)s/%(prefix)s/%(src_pkg)s" \ 

1258 "/%(src_pkg)s_%(src_ver)s/changelog" 

1259 elif self.candidate.origins[0].origin == "Ubuntu": 

1260 uri = "http://changelogs.ubuntu.com/changelogs/pool" \ 

1261 "/%(src_section)s/%(prefix)s/%(src_pkg)s" \ 

1262 "/%(src_pkg)s_%(src_ver)s/changelog" 

1263 else: 

1264 res = _("The list of changes is not available") 

1265 if isinstance(res, unicode): 

1266 return res 

1267 else: 

1268 return res.decode("utf-8") 

1269 

1270 # get the src package name 

1271 src_pkg = self.candidate.source_name 

1272

1274 src_section = "main" 

1275 # use the section of the candidate as a starting point 

1276 section = self.candidate.section 

1277

1279 src_ver = self.candidate.source_version

1283 # for some (e.g. libnspr4 on ubuntu) 

1284 # this feature only works if the correct deb-src are in the 

1285 # sources.list otherwise we fall back to the binary version number 

1286 src_records = apt_pkg.SourceRecords() 

1287 except SystemError: 

1288 pass

1294 # Direct match, use it and do not do more lookups. 

1295 src_ver = src_records.version 

1296 section = src_records.section 

1297 break

1299 # The version is higher, it seems to match. 

1300 src_ver = src_records.version 

1301 section = src_records.section 

1302

1304 if len(section_split) > 1: 

1305 src_section = section_split[0] 

1306 del section_split

1309 prefix = src_pkg[0] 

1310 if src_pkg.startswith("lib"): 

1311 prefix = "lib" + src_pkg[3] 

1312 

1313 # stip epoch 

1314 src_ver_split = src_ver.split(":", 1) 

1315 if len(src_ver_split) > 1: 

1316 src_ver = "".join(src_ver_split[1:]) 

1317 del src_ver_split

1320 "prefix": prefix, 

1321 "src_pkg": src_pkg, 

1322 "src_ver": src_ver}

1325 

1326 # FIXME: when python2.4 vanishes from the archive, 

1327 # merge this into a single try..finally block (pep 341)

1343 if cancel_lock and cancel_lock.is_set(): 

1344 return u""

1352

1357 # and from changelog too 

1358 installed = getattr(self.installed, 'version', None) 

1359 if installed and ":" in installed: 

1360 installed = installed.split(":", 1)[1]

1362 if changelog_ver and ":" in changelog_ver:

1366 changelog_ver, installed) <= 0):

1369 changelog += line 

1370

1372 if len(changelog) == 0: 

1373 changelog = _("The list of changes is not available") 

1374 if not isinstance(changelog, unicode):

1377 

1378 except HTTPError: 

1379 if self.candidate.origins[0].origin == "Ubuntu": 

1380 res = _("The list of changes is not available yet.\n\n" 

1381 "Please use " 

1382 "http://launchpad.net/ubuntu/+source/%s/" 

1383 "%s/+changelog\n" 

1384 "until the changes become available or try again " 

1385 "later.") % (src_pkg, src_ver) 

1386 else: 

1387 res = _("The list of changes is not available") 

1388 if isinstance(res, unicode): 

1389 return res 

1390 else: 

1391 return res.decode("utf-8") 

1392 except (IOError, BadStatusLine):

1395 if isinstance(res, unicode):
