178 def run(self, obj): 

179 """Install using the object 'obj'. 

180 

181 This functions runs install actions. The parameter 'obj' may either 

182 be a PackageManager object in which case its do_install() method is 

183 called or the path to a deb file. 

184 

185 If the object is a PackageManager, the functions returns the result 

186 of calling its do_install() method. Otherwise, the function returns 

187 the exit status of dpkg. In both cases, 0 means that there were no 

188 problems. 

189 """ 

190 pid = self.fork() 

191 if pid == 0: 

192 # pm.do_install might raise a exception, 

193 # when this happens, we need to catch 

194 # it, otherwise os._exit() is not run 

195 # and the execution continues in the 

196 # parent code leading to very confusing bugs 

197 try: 

198 os._exit(obj.do_install(self.write_stream.fileno())) 

199 except AttributeError: 

200 os._exit(os.spawnlp(os.P_WAIT, "dpkg", "dpkg", "--status-fd", 

201 str(self.write_stream.fileno()), "-i", 

202 obj)) 

203 except Exception as e: 

204 sys.stderr.write("%s\n" % e) 

205 os._exit(apt_pkg.PackageManager.RESULT_FAILED) 

206 

207 self.child_pid = pid 

208 res = self.wait_child() 

209 return os.WEXITSTATUS(res)

215 def update_interface(self): 

216 """Update the interface.""" 

217 try: 

218 line = self.status_stream.readline() 

219 except IOError as err: 

220 # resource temporarly unavailable is ignored 

221 if err.errno != errno.EAGAIN and err.errno != errno.EWOULDBLOCK: 

222 print(err.strerror) 

223 return 

224 

225 pkgname = status = status_str = percent = base = "" 

226 

227 if line.startswith('pm'): 

228 try: 

229 (status, pkgname, percent, status_str) = line.split(":", 3) 

230 except ValueError: 

231 # silently ignore lines that can't be parsed 

232 return 

233 elif line.startswith('status'): 

234 try: 

235 (base, pkgname, status, status_str) = line.split(":", 3) 

236 except ValueError: 

237 (base, pkgname, status) = line.split(":", 2) 

238 elif line.startswith('processing'): 

239 (status, status_str, pkgname) = line.split(":", 2) 

240 self.processing(pkgname.strip(), status_str.strip()) 

241 

242 # Always strip the status message 

243 pkgname = pkgname.strip() 

244 status_str = status_str.strip() 

245 status = status.strip() 

246 

247 if status == 'pmerror' or status == 'error': 

248 self.error(pkgname, status_str) 

249 elif status == 'conffile-prompt' or status == 'pmconffile': 

250 match = re.match("\s*\'(.*)\'\s*\'(.*)\'.*", status_str) 

251 if match: 

252 self.conffile(match.group(1), match.group(2)) 

253 elif status == "pmstatus": 

254 # FIXME: Float comparison 

255 if float(percent) != self.percent or status_str != self.status: 

256 self.status_change(pkgname, float(percent), status_str.strip()) 

257 self.percent = float(percent) 

258 self.status = status_str.strip() 

259 elif base == "status": 

260 self.dpkg_status_change(pkgname, status)

262 def wait_child(self): 

263 """Wait for child progress to exit. 

264 

265 This method is responsible for calling update_interface() from time to 

266 time. It exits once the child has exited. The return values is the 

267 full status returned from os.waitpid() (not only the return code). 

268 """ 

269 (pid, res) = (0, 0) 

270 while True: 

271 try: 

272 select.select([self.status_stream], [], [], 

273 self.select_timeout) 

274 except select.error as error: 

275 (errno_, _errstr) = error.args 

276 if errno_ != errno.EINTR: 

277 raise 

278 

279 self.update_interface() 

280 try: 

281 (pid, res) = os.waitpid(self.child_pid, os.WNOHANG) 

282 if pid == self.child_pid: 

283 break 

284 except OSError as err: 

285 if err.errno == errno.ECHILD: 

286 break 

287 if err.errno != errno.EINTR: 

288 raise 

289 

290 return res
