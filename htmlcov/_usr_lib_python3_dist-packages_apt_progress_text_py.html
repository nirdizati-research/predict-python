34def _(msg): 

35 """Translate the message, also try apt if translation is missing.""" 

36 res = apt_pkg.gettext(msg) 

37 if res == msg: 

38 res = apt_pkg.gettext(msg, "apt") 

39 return res

42class TextProgress(object): 

43 """Internal Base class for text progress classes.""" 

44 

45 def __init__(self, outfile=None): 

46 self._file = outfile or sys.stdout 

47 self._width = 0 

48 

49 def _write(self, msg, newline=True, maximize=False): 

50 """Write the message on the terminal, fill remaining space.""" 

51 self._file.write("\r") 

52 self._file.write(msg) 

53 

54 # Fill remaining stuff with whitespace 

55 if self._width > len(msg): 

56 self._file.write((self._width - len(msg)) * ' ') 

57 elif maximize: # Needed for OpProgress. 

58 self._width = max(self._width, len(msg)) 

59 if newline: 

60 self._file.write("\n") 

61 else: 

62 #self._file.write("\r") 

63 self._file.flush()

93class AcquireProgress(base.AcquireProgress, TextProgress): 

94 """AcquireProgress for the text interface."""

96 def __init__(self, outfile=None): 

97 TextProgress.__init__(self, outfile) 

98 base.AcquireProgress.__init__(self) 

99 self._signal = None 

100 self._width = 80 

101 self._id = long(1)

103 def start(self): 

104 """Start an Acquire progress. 

105 

106 In this case, the function sets up a signal handler for SIGWINCH, i.e. 

107 window resize signals. And it also sets id to 1. 

108 """ 

109 base.AcquireProgress.start(self) 

110 import signal 

111 self._signal = signal.signal(signal.SIGWINCH, self._winch) 

112 # Get the window size. 

113 self._winch() 

114 self._id = long(1) 

115 

116 def _winch(self, *dummy): 

117 """Signal handler for window resize signals.""" 

118 if hasattr(self._file, "fileno") and os.isatty(self._file.fileno()): 

119 import fcntl 

120 import termios 

121 import struct 

122 buf = fcntl.ioctl(self._file, termios.TIOCGWINSZ, 8 * ' ') 

123 dummy, col, dummy, dummy = struct.unpack('hhhh', buf) 

124 self._width = col - 1 # 1 for the cursor 

125 

126 def ims_hit(self, item): 

127 """Called when an item is update (e.g. not modified on the server).""" 

128 base.AcquireProgress.ims_hit(self, item) 

129 line = _('Hit ') + item.description 

130 if item.owner.filesize: 

131 line += ' [%sB]' % apt_pkg.size_to_str(item.owner.filesize) 

132 self._write(line) 

133 

134 def fail(self, item): 

135 """Called when an item is failed.""" 

136 base.AcquireProgress.fail(self, item) 

137 if item.owner.status == item.owner.STAT_DONE: 

138 self._write(_("Ign ") + item.description) 

139 else: 

140 self._write(_("Err ") + item.description) 

141 self._write(" %s" % item.owner.error_text) 

142 

143 def fetch(self, item): 

144 """Called when some of the item's data is fetched.""" 

145 base.AcquireProgress.fetch(self, item) 

146 # It's complete already (e.g. Hit) 

147 if item.owner.complete: 

148 return 

149 item.owner.id = self._id 

150 self._id += 1 

151 line = _("Get:") + "%s %s" % (item.owner.id, item.description) 

152 if item.owner.filesize: 

153 line += (" [%sB]" % apt_pkg.size_to_str(item.owner.filesize)) 

154 

155 self._write(line)

157 def pulse(self, owner): 

158 """Periodically invoked while the Acquire process is underway. 

159 

160 Return False if the user asked to cancel the whole Acquire process.""" 

161 base.AcquireProgress.pulse(self, owner) 

162 # only show progress on a tty to not clutter log files etc 

163 if (hasattr(self._file, "fileno") and 

164 not os.isatty(self._file.fileno())): 

165 return True 

166 

167 # calculate progress 

168 percent = (((self.current_bytes + self.current_items) * 100.0) / 

169 float(self.total_bytes + self.total_items)) 

170 

171 shown = False 

172 tval = '%i%%' % percent 

173 end = "" 

174 if self.current_cps: 

175 eta = long(float(self.total_bytes - self.current_bytes) / 

176 self.current_cps) 

177 end = " %sB/s %s" % (apt_pkg.size_to_str(self.current_cps), 

178 apt_pkg.time_to_str(eta)) 

179 

180 for worker in owner.workers: 

181 val = '' 

182 if not worker.current_item: 

183 if worker.status: 

184 val = ' [%s]' % worker.status 

185 if len(tval) + len(val) + len(end) >= self._width: 

186 break 

187 tval += val 

188 shown = True 

189 continue 

190 shown = True 

191 

192 if worker.current_item.owner.id: 

193 val += " [%i %s" % (worker.current_item.owner.id, 

194 worker.current_item.shortdesc) 

195 else: 

196 val += ' [%s' % worker.current_item.description 

197 if worker.current_item.owner.active_subprocess: 

198 val += ' %s' % worker.current_item.owner.active_subprocess 

199 

200 val += ' %sB' % apt_pkg.size_to_str(worker.current_size) 

201 

202 # Add the total size and percent 

203 if worker.total_size and not worker.current_item.owner.complete: 

204 val += "/%sB %i%%" % ( 

205 apt_pkg.size_to_str(worker.total_size), 

206 worker.current_size * 100.0 / worker.total_size) 

207 

208 val += ']' 

209 

210 if len(tval) + len(val) + len(end) >= self._width: 

211 # Display as many items as screen width 

212 break 

213 else: 

214 tval += val 

215 

216 if not shown: 

217 tval += _(" [Working]") 

218 

219 if self.current_cps: 

220 tval += (self._width - len(end) - len(tval)) * ' ' + end 

221 

222 self._write(tval, False) 

223 return True
